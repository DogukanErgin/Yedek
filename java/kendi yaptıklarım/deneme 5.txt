public class Calisan{
    private String isim;
    private String departman;
    private int maas;
    public Calisan(String isim,String departman,int maas){
        this.isim=isim;
        this.departman=departman;
        this.maas=maas;
    }
    public void zamyap(int zammiktarý){

        this.maas=maas+zammiktarý;
        System.out.println(zammiktarý+"? zam yapildi"+"yeni maasiniz = "+maas);
    }
    public void departmandegis(String yenidepartman){
        this.departman=yenidepartman;
    }



}

}
    }
